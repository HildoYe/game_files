	key = "KALEIDOSCOPE_INFESTATION"
	cost = 500
	tech_department = engineering_technology

	icon = "gfx/interface/icons/situation_log/situation_log_engineering.dds"
	location = yes
	event_scope = ship_event

		shipclass_constructor = 1
	}

			}
		}
	}
